# Mersenne twister算法介绍

在Python中random库的默认伪随机数生成器(PRNG)采用了**梅森旋转算法**(Mersenne Twister),该算法是基于有限二进制字段上的矩阵线性递归 $\mathbb{F}_2$ ,可以高效产生高质量的伪随机数,这弥补了传统伪随机数生成器的不足.该算法的最长周期取自一个梅森素数 $2^{19937}-1$ ,故也被称为**MT19937**

## 代码实现

该算法主要分为三个阶段

- 第一阶段:获得基础的梅森旋转链(初始化)
- 第二阶段:对于旋转链进行旋转算法
- 第三阶段:对于旋转算法所得的结果进行处理

使用MT19937算法生成范围在 $[0,2^{32}−1]$ 的均匀分布的 $32$ 位无符号整数

```python
# 将数字变为32位整数
def _int32(x):
    # d=0xffffffff 额外梅森旋转所需的掩码和位移量
    return int(0xffffffff & x)

class MT19937:
    # 第一阶段:初始化
    def __init__(self, seed):
        self.mt = [0] * 624
        self.mt[0] = seed
        self.mti = 0
        for i in range(1, 624):
            # f=0x6c078965 初始化梅森旋转链所需参数
            self.mt[i] = _int32(0x6c078965 * (self.mt[i - 1] ^ self.mt[i - 1] >> 30) + i)

    # 第三阶段:提取伪随机数
    def extract_number(self):
        if self.mti == 0:
            self.twist()
        y = self.mt[self.mti]
        # u=11 额外梅森旋转所需的掩码和位移量
        y = y ^ (y >> 11)
        # s=7 TGFSR的位移量  b=0x9d2c5680 TGFSR的掩码
        y = y ^ ((y << 7) & 0x9d2c5680)
        # t=15 TGFSR的位移量 c=0xefc60000 TGFSR的掩码
        y = y ^ ((y << 15) & 0xefc60000)
        # l=18 额外梅森旋转所需的掩码和位移量
        y = y ^ (y >> 18)
        self.mti = (self.mti + 1) % 624
        return _int32(y)

    # 第二阶段:对状态旋转
    def twist(self):
        for i in range(0, 624):
            # kUpperBits=0x80000000 用于提取32位整数的最高位
            # kLowerBits=0x7fffffff 用于提取32位整数的低31位
            y = _int32((self.mt[i] & 0x80000000) + (self.mt[(i + 1) % 624] & 0x7fffffff))
            # m=397 周期参数,用作第三阶段的偏移量 n=624 递归长度
            self.mt[i] = (y >> 1) ^ self.mt[(i + 397) % 624]

            if y % 2 != 0:
                # a=0x9908b0df 旋转矩阵的参数
                self.mt[i] = self.mt[i] ^ 0x9908b0df

if __name__ == "__main__":
    seed = 12345  # 初始种子
    mt = MT19937(seed)
    for _ in range(10):
        print(mt.extract_number())
```

## 题型一：逆向随机数生成器内部的state状态

分析extract_number函数可以发现该函数对state\[i]进行了一系列的异或和位运算,最后输出结果
首先分析如下表达式,以下计算都带入一个随机的 $32$ 位数

```python
y = y ^ (y >> 18)
```

发现这个表达式并没有丢失掉其高 $18$ 位,它只是将原高位移至原低位处并异或 $(1 ＾ 1 = 0)$ ,结合异或运算的可逆性

```python
o = 2299897386
print(bin(o)[2:])
#10001001000101011010011000101010
o_high = o >> 18
print(bin(o_high)[2:].zfill(32))
#00000000000000000010001001000101
y = o ^ o_high
print(bin(y)[2:])
#10001001000101011000010001101111
for i in range(32//18):
    y_high = y >> 18
    print(bin(y_high)[2:].zfill(32))
    #00000000000000000010001001000101
    y = y ^ y_high
print(y==o)
```

接着分析

```python
y = y ^ ((y << 15) & 0xefc60000)
```

这个表达式首先是在 $y$ 的低位后面直接加了 $15$ 个 $0$ ,得到的值再与 $0xefc60000$ 按位与 $(1 \& 1 = 1)$ ,最后又与原 $y$ 异或

```python
o = 2299897386
print(bin(o)[2:].zfill(47))
#00000000000000010001001000101011010011000101010
o_low = o << 15
print(bin(o_low)[2:])
#10001001000101011010011000101010000000000000000
bitand = o_low & 0xefc60000
print(bin(0xefc60000)[2:].zfill(47))
#00000000000000011101111110001100000000000000000
print(bin(bitand)[2:].zfill(47))
#00000000000000011000011000001000000000000000000
y = o ^ bitand
print(bin(y)[2:].zfill(47))
#00000000000000001001010000100011010011000101010
tmp = y
for i in range(32 // 15):
    # 每次可恢复y的15位
    y = tmp ^ ((y << 15) & 0xefc60000)
#print(bin(y_low)[2:].zfill(47))
#01001010000100011010011000101010000000000000000
print(y==o)
```

同理,剩下两步就不再赘述.完整逆向代码如下

```python
o = 2299897386

def inverse_right_mask(res, shift, mask=0xffffffff, bits=32):
    tmp = res
    for i in range(bits // shift):
        tmp = res ^ tmp >> shift & mask
    return tmp

def inverse_left_mask(res, shift, mask=0xffffffff, bits=32):
    tmp = res
    for i in range(bits // shift):
        tmp = res ^ tmp << shift & mask
    return tmp

def extract_number(y):
    y = y ^ y >> 11
    y = y ^ y << 7 & 0x9d2c5680
    y = y ^ y << 15 & 0xefc60000
    y = y ^ y >> 18
    return y & 0xffffffff

def recover(y):
    y = inverse_right_mask(y,18)
    y = inverse_left_mask(y,15,0xefc60000)
    y = inverse_left_mask(y,7,0x9d2c5680)
    y = inverse_right_mask(y,11)
    return y & 0xffffffff
print(recover(extract_number(o)) == o)
```

def circle(m):
    t = m

from Crypto.Util.number import bytes_to_long, long_to_bytes

# right shift inverse
def inverse_right(res, shift, bits=32):
    tmp = res
    for i in range(bits // shift):
        tmp = res ^ tmp >> shift
    return tmp

# right shift with mask inverse
def inverse_right_mask(res, shift, mask, bits=32):
    tmp = res
    for i in range(bits // shift):
        tmp = res ^ tmp >> shift & mask
    return tmp

# left shift inverse
def inverse_left(res, shift, bits=32):
    tmp = res
    for i in range(bits // shift):
        tmp = res ^ tmp << shift
    return tmp

# left shift with mask inverse
def inverse_left_mask(res, shift, mask, bits=32):
    tmp = res
    for i in range(bits // shift):
        tmp = res ^ tmp << shift & mask
    return tmp

    y = y ^ y << 7 & 2636928640

from hashlib import md5
from Crypto.Cipher import AES
import binascii
from randcrack import RandCrack
from Crypto.Util.Padding import unpad

with open('D:\\output.txt','r') as f:
    lines = [int(line) for line in f.readlines()]
    f.close()

rc = RandCrack()
for i in range(len(lines)-624,666):
    rc.submit(lines[i])
r = rc.predict_getrandbits(32)
print(r)
#2496065772
key = bytes.fromhex(hex(r)[2:])
key = md5(key).digest()
c = binascii.unhexlify("91fd1824ddc0a35e845e87f59e53a103334df418e6a65a7d7769699c3ca2119019361cd23a46a61d4e7f6cdff5f5200f586f90b66eabfd8ecff4ddf11ee444d37f80ada0bbe8af09e4fc32c1a51e3f29e2771b51c71d2ba4acb84fda61904b96")

c = c[16:]
aes = AES.new(key,AES.MODE_CBC,iv)
m = aes.decrypt(c)
print(unpad(m,16))
